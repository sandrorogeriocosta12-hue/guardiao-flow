let visitasAtivas = {}; document.addEventListener('DOMContentLoaded', function() { const params = new URLSearchParams(window.location.search); const tokenUrl = params.get('token'); if (tokenUrl) { document.getElementById('morador-token').value = tokenUrl; setTimeout(() => carregarVisitasPendentes(tokenUrl), 500); } document.getElementById('btn-carregar-visitas').addEventListener('click', () => { const token = document.getElementById('morador-token').value.trim(); if (token) { carregarVisitasPendentes(token); } else { mostrarToast('Por favor, insira seu token', 'error'); } }); document.getElementById('morador-token').addEventListener('keypress', (e) => { if (e.key === 'Enter') { document.getElementById('btn-carregar-visitas').click(); } }); atualizarStatusConexao(false); }); function atualizarStatusConexao(conectado) { const badge = document.getElementById('connection-status'); const statusText = document.getElementById('status-badge'); if (conectado) { badge.classList.add('connected'); statusText.textContent = moradorToken ? 'Conectado' : 'Conectado'; } else { badge.classList.remove('connected'); statusText.textContent = 'Offline'; } } async function carregarVisitasPendentes(token) { moradorToken = token; const btn = document.getElementById('btn-carregar-visitas'); btn.disabled = true; document.getElementById('btn-text').innerHTML = ' Carregando...'; try { const response = await fetch(`${API_HOST}/api/morador/visitas_pendentes/${token}`); if (!response.ok) { throw new Error('Token inválido'); } const visitas = await response.json(); visitasAtivas = {}; document.getElementById('visitas-lista').innerHTML = ''; document.getElementById('token-section').classList.add('hidden'); document.getElementById('visitas-container').classList.add('show'); visitas.forEach(v => adicionarVisita(v)); atualizarStatusConexao(true); if (visitas.length === 0) { document.getElementById('empty-state').style.display = 'block'; } } catch (error) { console.error('Erro ao carregar visitas:', error); mostrarToast('Erro: ' + error.message, 'error'); document.getElementById('token-section').classList.remove('hidden'); document.getElementById('visitas-container').classList.remove('show'); } finally { btn.disabled = false; document.getElementById('btn-text').innerHTML = 'Carregar Visitantes Pendentes'; } } function adicionarVisita(visita) { if (visitasAtivas[visita.id]) return; visitasAtivas[visita.id] = visita; const card = document.createElement('div'); card.className = 'visita-card'; card.id = `visita-${visita.id}`; const agora = new Date(); const entrada = new Date(visita.horario_entrada); const minutos = Math.floor((agora - entrada) / 60000); const tempoTexto = minutos === 0 ? 'agora' : `há ${minutos}m`; card.innerHTML = `
${visita.nome}
${tempoTexto}
Placa ${visita.placa || 'Não informada'}
Destino Casa ${visita.destino}
`; card.querySelector('.btn-liberar').addEventListener('click', () => { autorizarVisita(visita.id, card); }); card.querySelector('.btn-rejeitar').addEventListener('click', () => { rejeitarVisita(visita.id, card); }); document.getElementById('visitas-lista').appendChild(card); document.getElementById('empty-state').style.display = 'none'; } function removerVisita(visitaId) { const card = document.getElementById(`visita-${visitaId}`); if (card) { card.style.animation = 'slideIn 0.3s ease reverse'; setTimeout(() => card.remove(), 300); } delete visitasAtivas[visitaId]; if (Object.keys(visitasAtivas).length === 0) { document.getElementById('empty-state').style.display = 'block'; } } async function autorizarVisita(visitaId, card) { const token = moradorToken; const botoes = card.querySelectorAll('.btn-acao'); botoes.forEach(b => b.disabled = true); try { const response = await fetch(`${API_HOST}/api/morador/liberar`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ visita_id: visitaId, token }) }); if (!response.ok) throw new Error('Erro ao autorizar'); const result = await response.json(); if (result.status === 'liberado') { mostrarToast('✓ Visitante autorizado!', 'success'); removerVisita(visitaId); } } catch (error) { console.error('Erro:', error); mostrarToast('Erro: ' + error.message, 'error'); botoes.forEach(b => b.disabled = false); } } async function rejeitarVisita(visitaId, card) { const token = moradorToken; const botoes = card.querySelectorAll('.btn-acao'); botoes.forEach(b => b.disabled = true); try { const response = await fetch(`${API_HOST}/api/morador/rejeitar`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ visita_id: visitaId, token }) }); if (!response.ok) throw new Error('Erro ao rejeitar'); const result = await response.json(); if (result.status === 'rejeitado') { mostrarToast('✕ Visitante recusado', 'success'); removerVisita(visitaId); } } catch (error) { console.error('Erro:', error); mostrarToast('Erro: ' + error.message, 'error'); botoes.forEach(b => b.disabled = false); } } function mostrarToast(mensagem, tipo = 'info') { const toast = document.createElement('div'); toast.className = `toast ${tipo}`; toast.textContent = mensagem; document.body.appendChild(toast); setTimeout(() => { toast.style.animation = 'slideUp 0.3s ease reverse'; setTimeout(() => toast.remove(), 300); }, 3000); }
